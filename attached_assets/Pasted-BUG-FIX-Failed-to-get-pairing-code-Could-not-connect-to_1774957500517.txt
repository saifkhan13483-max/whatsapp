BUG FIX: "Failed to get pairing code — Could not connect to WhatsApp"
Target: WaTracker Pro v2.0 — WhatsApp Pairing Code Connection Feature

═══════════════════════════════════════════════════════════════════
PROBLEM ANALYSIS
═══════════════════════════════════════════════════════════════════

The error "Failed to get pairing code / Could not connect to WhatsApp"
is shown on the frontend when POST /api/whatsapp/request-pairing-code fails.

This is caused by one or more of the following ROOT CAUSES on the backend:

ROOT CAUSE 1 — TIMING BUG (Most Common):
  sock.requestPairingCode() is being called IMMEDIATELY after makeWASocket(),
  before the socket has had time to connect to WhatsApp servers and emit
  either a 'qr' or 'connecting' event.
  Calling requestPairingCode() too early causes a 428 "Precondition Required"
  or "Connection Closed" error from WhatsApp servers.

  FIX: requestPairingCode() MUST be called INSIDE the 'connection.update'
  event listener, only after (connection === 'connecting') OR (!!qr) is true.
  NEVER call it immediately after makeWASocket() returns.

ROOT CAUSE 2 — PHONE NUMBER FORMAT:
  The phone number is passed with special characters (+, spaces, dashes,
  parentheses) or with the leading + sign.
  Baileys requires the phone number in pure digits only, no plus sign.
  Example: "+92 300 1234567" MUST become "923001234567" before passing
  to requestPairingCode().

  FIX: Always strip ALL non-digit characters from the phone number
  before calling requestPairingCode():
    const cleanNumber = phoneNumber.replace(/\D/g, '')

ROOT CAUSE 3 — printQRInTerminal NOT SET TO FALSE:
  If makeWASocket is called without printQRInTerminal: false, the socket
  may behave differently and interfere with the pairing code flow.

  FIX: Always set printQRInTerminal: false in makeWASocket config
  when using pairing code method.

ROOT CAUSE 4 — SESSION REUSE WITHOUT CLEANUP:
  If a previous broken/partial session exists in the database or memory
  for this user, and it is reused without being cleared, WhatsApp rejects
  the new pairing code request because creds.registered may be stale true.

  FIX: Before creating a new socket for a fresh pairing code request,
  destroy any existing socket for this user and clear their stored
  sessionData in the database.

ROOT CAUSE 5 — MISSING ERROR PROPAGATION:
  The backend catches the Baileys error but sends a generic 500 response
  without the actual cause, making debugging impossible and showing the
  user a vague "Could not connect" message.

  FIX: Parse the Baileys/Boom error statusCode and map it to a
  human-readable message before returning the response.

ROOT CAUSE 6 — SOCKET DESTROYED BEFORE CODE IS RETURNED:
  The Promise resolving requestPairingCode() returns the code, but if the
  socket emits a 'connection.update' with connection === 'close' before
  the HTTP response is sent, the code is lost.

  FIX: Wrap the entire flow in a Promise that resolves on code received
  and rejects on connection close, with a timeout of 30 seconds.

// In your SessionManager or route handler:

async function requestPairingCodeForUser(
  userId: number,
  rawPhoneNumber: string
): Promise<{ pairingCode: string; expiresAt: string }> {

  // STEP 1: Sanitize phone number — digits only, no +, no spaces, no dashes
  const phoneNumber = rawPhoneNumber.replace(/\D/g, '')

  if (!phoneNumber || phoneNumber.length < 7) {
    throw new Error('INVALID_PHONE_NUMBER')
  }

  // STEP 2: Destroy any existing socket for this user
  // to prevent stale session interference
  const existingSocket = activeSockets.get(userId)
  if (existingSocket) {
    try { existingSocket.end(undefined) } catch (_) {}
    activeSockets.delete(userId)
  }

  // STEP 3: Clear stored session from DB so creds.registered is fresh
  await db.update(whatsappSessions)
    .set({ sessionData: null, status: 'pending_pairing' })
    .where(eq(whatsappSessions.userId, userId))

  // STEP 4: Get latest Baileys version
  const { version } = await fetchLatestBaileysVersion()

  // STEP 5: Initialize auth state (load from DB or create fresh)
  // Use your DB-backed auth state, NOT useMultiFileAuthState
  const { state, saveCreds } = await getDbAuthState(userId)

  // STEP 6: Create socket with REQUIRED config for pairing code
  const sock = makeWASocket({
    version,
    auth: {
      creds: state.creds,
      keys: makeCacheableSignalKeyStore(state.keys, logger),
    },
    printQRInTerminal: false,  // REQUIRED — must be false for pairing code
    logger,
    browser: ['WaTracker Pro', 'Chrome', '120.0.0'],
    syncFullHistory: false,
    generateHighQualityLinkPreview: false,
  })

  // Store socket in memory map
  activeSockets.set(userId, sock)

  // STEP 7: Save credentials whenever they update
  sock.ev.on('creds.update', saveCreds)

  // STEP 8: Wrap entire pairing code flow in a Promise with timeout
  return new Promise((resolve, reject) => {

    const TIMEOUT_MS = 30000  // 30 second timeout
    let codeRequested = false

    const timeout = setTimeout(() => {
      cleanup()
      reject(new Error('TIMEOUT'))
    }, TIMEOUT_MS)

    function cleanup() {
      clearTimeout(timeout)
      sock.ev.removeAllListeners('connection.update')
    }

    // STEP 9: Listen for connection.update BEFORE calling requestPairingCode
    // requestPairingCode MUST be called inside this event, not before it
    sock.ev.on('connection.update', async (update) => {
      const { connection, lastDisconnect, qr } = update

      // CRITICAL: Only call requestPairingCode when socket is
      // in 'connecting' state OR when QR is available
      // This is the correct timing — NOT immediately after makeWASocket
      if ((connection === 'connecting' || !!qr) && !codeRequested) {
        codeRequested = true

        try {
          const code = await sock.requestPairingCode(phoneNumber)

          if (!code) {
            cleanup()
            reject(new Error('EMPTY_CODE'))
            return
          }

          // Calculate expiry (WhatsApp pairing codes expire in ~5 minutes)
          const expiresAt = new Date(Date.now() + 5 * 60 * 1000).toISOString()

          // Persist to DB
          await db.update(whatsappSessions).set({
            pairingCode: code,
            pairingCodeExpiresAt: new Date(expiresAt),
            status: 'pending_pairing',
            phoneNumber: phoneNumber,
            maskedPhone: maskPhoneNumber(phoneNumber),
          }).where(eq(whatsappSessions.userId, userId))

          cleanup()
          resolve({ pairingCode: code, expiresAt })

        } catch (err: any) {
          cleanup()
          reject(err)
        }
      }

      // Handle connection closed before code was issued
      if (connection === 'close' && !codeRequested) {
        const statusCode = (lastDisconnect?.error as Boom)?.output?.statusCode
        cleanup()
        reject(new Error(`CONNECTION_CLOSED_${statusCode ?? 'UNKNOWN'}`))
      }

      // Handle successful full connection (pairing accepted)
      if (connection === 'open') {
        await db.update(whatsappSessions).set({
          status: 'connected',
          connectedAt: new Date(),
        }).where(eq(whatsappSessions.userId, userId))
      }
    })
  })
}

router.post('/whatsapp/request-pairing-code', requireAuth, async (req, res) => {
  try {
    const { phoneNumber } = req.body
    const userId = req.user.id

    if (!phoneNumber) {
      return res.status(400).json({ error: 'Phone number is required' })
    }

    const result = await requestPairingCodeForUser(userId, phoneNumber)

    return res.json({
      pairingCode: result.pairingCode,
      expiresAt: result.expiresAt,
    })

  } catch (err: any) {
    const message = err.message || ''

    // Map internal error codes to user-friendly messages
    if (message === 'INVALID_PHONE_NUMBER') {
      return res.status(400).json({
        error: 'Invalid phone number. Please include your country code.'
      })
    }
    if (message.includes('CONNECTION_CLOSED_401') || message.includes('401')) {
      return res.status(401).json({
        error: 'WhatsApp rejected the connection. Please try again.'
      })
    }
    if (message.includes('CONNECTION_CLOSED_428') || message.includes('428')) {
      return res.status(428).json({
        error: 'Connection was not ready. Please wait a moment and try again.'
      })
    }
    if (message === 'TIMEOUT') {
      return res.status(408).json({
        error: 'Request timed out. Please check your internet connection and try again.'
      })
    }
    if (message === 'EMPTY_CODE') {
      return res.status(502).json({
        error: 'WhatsApp returned an empty code. Please try again.'
      })
    }

    console.error('[WhatsApp] requestPairingCode error:', err)
    return res.status(500).json({
      error: 'Could not connect to WhatsApp. Please try again in a few seconds.'
    })
  }
})

Copyfunction maskPhoneNumber(phone: string): string {
  // phone is digits only e.g. "923001234567"
  if (phone.length < 7) return '***'
  const last4 = phone.slice(-4)
  const prefix = phone.slice(0, phone.length - 7)
  return `+${prefix} *** ${last4}`
}
